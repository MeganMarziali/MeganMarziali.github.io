data("ny_noaa")

sample_noaa = sample_n(ny_noaa, 500)

noaa_clean = 
  sample_noaa %>% 
  janitor::clean_names() %>% 
  separate(date, into = c("year", "month", "day"), convert = TRUE) %>% 
  mutate(
    tmax = as.numeric(tmax) / 10,
    tmin = as.numeric(tmin) / 10,
    snow_mm = snow,
    snow = round(snow_mm * 0.03937 * 4) / 4,
    prcp_mm = prcp,
    prcp = round(prcp_mm * 0.03937 * 4) / 4)
